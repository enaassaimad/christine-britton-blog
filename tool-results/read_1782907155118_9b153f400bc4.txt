     1→     1→'use client'

     3→     3→import { useEffect, useState, useRef } from 'react'
     4→     4→import { api } from '@/lib/api'
     5→     5→import type { Post, Category, Media, AffiliateLink } from '@/lib/types'
     6→     6→import { useApp } from '@/store/app-store'
     7→     7→import { Button } from '@/components/ui/button'
     8→     8→import { Input } from '@/components/ui/input'
     9→     9→import { Textarea } from '@/components/ui/textarea'
    10→    10→import { Label } from '@/components/ui/label'
    11→    11→import { Switch } from '@/components/ui/switch'
    12→    12→import { Badge } from '@/components/ui/badge'
    13→    13→import { RadioGroup, RadioGroupItem } from '@/components/ui/radio-group'
    14→    14→import { Tabs, TabsContent, TabsList, TabsTrigger } from '@/components/ui/tabs'
    15→    15→import { Markdown } from '@/components/site/markdown'
    16→    16→import { Metabox } from './metabox'
    17→    17→import { AffiliateLinksEditor } from './affiliate-links-editor'
    18→    18→import { GenerateImageDialog } from './generate-image-dialog'
    19→    19→import { InsertHtmlDialog, InsertAffiliateDialog } from './insert-dialogs'
    20→    20→import {
    21→    21→  ArrowLeft, Save, Eye, Upload, ImageIcon, Loader2, Sparkles, Send, Clock,
    22→    22→  Calendar, Tag, FolderTree, Image as ImgIcon, FileText, Search, ShoppingCart,
    23→    23→  Settings as SettingsIcon, Wand2, Link2, Check, AlertCircle, Eye as EyeIcon,
    24→    24→  Code2,
    25→    25→} from 'lucide-react'
    26→    26→import { toast } from 'sonner'
    27→    27→import { slugify, excerptFromContent, estimateReadTime } from '@/lib/helpers'
    28→    28→
    29→    29→interface EditorState {
    30→    30→  title: string
    31→    31→  slug: string
    32→    32→  excerpt: string
    33→    33→  content: string
    34→    34→  coverImage: string
    35→    35→  coverAlt: string
    36→    36→  categoryId: string
    37→    37→  tags: string
    38→    38→  status: string
    39→    39→  featured: boolean
    40→    40→  trending: boolean
    41→    41→  showAds: boolean
    42→    42→  metaTitle: string
    43→    43→  metaDescription: string
    44→    44→  publishedAt: string
    45→    45→  affiliateLinks: AffiliateLink[]
    46→    46→}
    47→    47→
    48→    48→const BLANK: EditorState = {
    49→    49→  title: '', slug: '', excerpt: '', content: '', coverImage: '', coverAlt: '',
    50→    50→  categoryId: '', tags: '', status: 'DRAFT', featured: false, trending: false,
    51→    51→  showAds: true, metaTitle: '', metaDescription: '', publishedAt: '', affiliateLinks: [],
    52→    52→}
    53→    53→
    54→    54→export function PostEditor({ postId }: { postId: string | null }) {
    55→    55→  const { setAdminView, openPost } = useApp()
    56→    56→  const [categories, setCategories] = useState<Category[]>([])
    57→    57→  const [media, setMedia] = useState<Media[]>([])
    58→    58→  const [saving, setSaving] = useState(false)
    59→    59→  const [loading, setLoading] = useState(!!postId)
    60→    60→  const [showMedia, setShowMedia] = useState(false)
    61→    61→  const [uploading, setUploading] = useState(false)
    62→    62→  const [showGenerator, setShowGenerator] = useState(false)
    63→    63→  const [showHtmlDialog, setShowHtmlDialog] = useState(false)
    64→    64→  const [showAffiliateDialog, setShowAffiliateDialog] = useState(false)
    65→    65→  const [form, setForm] = useState<EditorState>(BLANK)
    66→    66→  const fileRef = useRef<HTMLInputElement>(null)
    67→    67→  const slugTouched = useRef(false)
    68→    68→  const contentRef = useRef<HTMLTextAreaElement>(null)
    69→    69→
    70→    70→  // Insert text at the cursor position in the content textarea
    71→    71→  const insertAtCursor = (text: string) => {
    72→    72→    const ta = contentRef.current
    73→    73→    if (!ta) {
    74→    74→      update({ content: form.content + '\n\n' + text })
    75→    75→      return
    76→    76→    }
    77→    77→    const start = ta.selectionStart
    78→    78→    const end = ta.selectionEnd
    79→    79→    const before = form.content.slice(0, start)
    80→    80→    const after = form.content.slice(end)
    81→    81→    const insert = (before.endsWith('\n') || before === '' ? '' : '\n\n') + text + '\n\n'
    82→    82→    const newContent = before + insert + after
    83→    83→    update({ content: newContent })
    84→    84→    // Restore cursor position after the inserted text
    85→    85→    requestAnimationFrame(() => {
    86→    86→      ta.focus()
    87→    87→      const pos = (start + insert.length) as number
    88→    88→      ta.setSelectionRange(pos, pos)
    89→    89→    })
    90→    90→  }
    91→    91→
    92→    92→  useEffect(() => {
    93→    93→    let active = true
    94→    94→    api.categories.list().then(({ categories }) => {
    95→    95→      if (!active) return
    96→    96→      setCategories(categories)
    97→    97→      if (!postId && categories[0]) setForm((f) => ({ ...f, categoryId: categories[0].id }))
    98→    98→    })
    99→    99→    api.media.list().then(({ media }) => { if (active) setMedia(media) })
   100→   100→    return () => { active = false }
   101→   101→  }, [postId])
   102→   102→
   103→   103→  useEffect(() => {
   104→   104→    if (!postId) { setLoading(false); return }
   105→   105→    setLoading(true)
   106→   106→    api.posts.get(postId).then(({ post }) => {
   107→   107→      setForm({
   108→   108→        title: post.title, slug: post.slug, excerpt: post.excerpt || '', content: post.content,
   109→   109→        coverImage: post.coverImage || '', coverAlt: post.coverAlt || '', categoryId: post.categoryId,
   110→   110→        tags: post.tags || '', status: post.status, featured: post.featured, trending: post.trending,
   111→   111→        showAds: post.showAds, metaTitle: post.metaTitle || '', metaDescription: post.metaDescription || '',
   112→   112→        publishedAt: post.publishedAt ? toLocalInput(post.publishedAt) : '',
   113→   113→        affiliateLinks: (post as any).affiliateLinks || [],
   114→   114→      })
   115→   115→    }).finally(() => setLoading(false))
   116→   116→  }, [postId])
   117→   117→
   118→   118→  const update = (patch: Partial<EditorState>) => setForm((f) => ({ ...f, ...patch }))
   119→   119→
   120→   120→  const onTitleChange = (title: string) => {
   121→   121→    if (!slugTouched.current) update({ title, slug: slugify(title) })
   122→   122→    else update({ title })
   123→   123→  }
   124→   124→
   125→   125→  const upload = async (file: File) => {
   126→   126→    setUploading(true)
   127→   127→    try {
   128→   128→      const { url } = await api.media.upload(file)
   129→   129→      setMedia((m) => [{ id: url, url, type: 'image', createdAt: new Date().toISOString() }, ...m])
   130→   130→      update({ coverImage: url })
   131→   131→      toast.success('Image uploaded.')
   132→   132→    } catch { toast.error('Upload failed.') } finally { setUploading(false) }
   133→   133→  }
   134→   134→
   135→   135→  // Convert an ISO date to a value usable in <input type="datetime-local">
   136→   136→  function toLocalInput(iso: string): string {
   137→   137→    try {
   138→   138→      const d = new Date(iso)
   139→   139→      const off = d.getTimezoneOffset()
   140→   140→      const local = new Date(d.getTime() - off * 60000)
   141→   141→      return local.toISOString().slice(0, 16)
   142→   142→    } catch { return '' }
   143→   143→  }
   144→   144→
   145→   145→  // Determine the scheduling state from status + date
   146→   146→  const scheduledDate = form.publishedAt ? new Date(form.publishedAt) : null
   147→   147→  const isScheduledFuture = form.status === 'SCHEDULED' && scheduledDate && scheduledDate > new Date()
   148→   148→
   149→   149→  const save = async (mode: 'draft' | 'publish' | 'schedule') => {
   150→   150→    if (!form.title.trim()) { toast.error('Title is required.'); return }
   151→   151→    if (!form.categoryId) { toast.error('Please choose a category.'); return }
   152→   152→    if (!form.content.trim()) { toast.error('Content cannot be empty.'); return }
   153→   153→    if (mode === 'schedule' && !form.publishedAt) {
   154→   154→      toast.error('Please set a publish date to schedule.')
   155→   155→      return
   156→   156→    }
   157→   157→    if (mode === 'schedule' && scheduledDate && scheduledDate <= new Date()) {
   158→   158→      toast.error('Schedule date must be in the future.')
   159→   159→      return
   160→   160→    }
   161→   161→
   162→   162→    let status = form.status
   163→   163→    if (mode === 'draft') status = 'DRAFT'
   164→   164→    else if (mode === 'publish') status = 'PUBLISHED'
   165→   165→    else if (mode === 'schedule') status = 'SCHEDULED'
   166→   166→
   167→   167→    setSaving(true)
   168→   168→    const data: any = {
   169→   169→      title: form.title, slug: form.slug || slugify(form.title),
   170→   170→      excerpt: form.excerpt || excerptFromContent(form.content),
   171→   171→      content: form.content, coverImage: form.coverImage || null, coverAlt: form.coverAlt,
   172→   172→      categoryId: form.categoryId, tags: form.tags, status,
   173→   173→      featured: form.featured, trending: form.trending, showAds: form.showAds,
   174→   174→      metaTitle: form.metaTitle || null, metaDescription: form.metaDescription || null,
   175→   175→      publishedAt: form.publishedAt || null,
   176→   176→      affiliateLinks: form.affiliateLinks,
   177→   177→    }
   178→   178→    try {
   179→   179→      let saved: Post
   180→   180→      if (postId) {
   181→   181→        const { post } = await api.posts.update(postId, data)
   182→   182→        saved = post
   183→   183→      } else {
   184→   184→        const { post } = await api.posts.create(data)
   185→   185→        saved = post
   186→   186→      }
   187→   187→      const verb = mode === 'publish' ? 'Published!' : mode === 'schedule' ? 'Scheduled!' : 'Saved.'
   188→   188→      toast.success(verb)
   189→   189→      setAdminView('posts')
   190→   190→      if (mode === 'publish') openPost(saved.slug)
   191→   191→    } catch (e: any) {
   192→   192→      toast.error(e.message || 'Save failed.')
   193→   193→    } finally { setSaving(false) }
   194→   194→  }
   195→   195→
   196→   196→  const insertImageIntoContent = (img: { url: string; prompt: string }) => {
   197→   197→    const md = `\n\n![${img.prompt.slice(0, 80)}](${img.url})\n\n`
   198→   198→    update({ content: form.content + md })
   199→   199→    toast.success('Image inserted into content.')
   200→   200→  }
   201→   201→
   202→   202→  if (loading) {
   203→   203→    return <div className="p-12 text-center"><Loader2 className="h-6 w-6 animate-spin mx-auto text-muted-foreground" /></div>
   204→   204→  }
   205→   205→
   206→   206→  const wordCount = form.content.split(/\s+/).filter(Boolean).length
   207→   207→  const statusBadge = form.status === 'PUBLISHED'
   208→   208→    ? { label: 'Published', cls: 'bg-emerald-600 text-white' }
   209→   209→    : form.status === 'SCHEDULED'
   210→   210→    ? { label: 'Scheduled', cls: 'bg-amber-500 text-white' }
   211→   211→    : { label: 'Draft', cls: 'bg-muted text-muted-foreground' }
   212→   212→
   213→   213→  return (
   214→   214→    <div className="p-4 md:p-6 max-w-7xl mx-auto">
   215→   215→      {/* Top bar — WordPress style */}
   216→   216→      <div className="flex flex-wrap items-center justify-between gap-3 mb-5">
   217→   217→        <div className="flex items-center gap-3">
   218→   218→          <Button variant="ghost" size="sm" onClick={() => setAdminView('posts')}>
   219→   219→            <ArrowLeft className="h-4 w-4 mr-1.5" /> Articles
   220→   220→          </Button>
   221→   221→          <Badge className={statusBadge.cls}>{statusBadge.label}</Badge>
   222→   222→          {form.title && <span className="text-sm text-muted-foreground truncate max-w-[300px]">{form.title}</span>}
   223→   223→        </div>
   224→   224→        <div className="flex items-center gap-2">
   225→   225→          <Button variant="outline" onClick={() => save('draft')} disabled={saving}>
   226→   226→            <Save className="h-4 w-4 mr-1.5" /> {saving ? 'Saving…' : 'Save draft'}
   227→   227→          </Button>
   228→   228→          {isScheduledFuture ? (
   229→   229→            <Button onClick={() => save('schedule')} disabled={saving} className="bg-amber-600 hover:bg-amber-700">
   230→   230→              <Clock className="h-4 w-4 mr-1.5" /> {saving ? 'Saving…' : 'Schedule'}
   231→   231→            </Button>
   232→   232→          ) : (
   233→   233→            <Button onClick={() => save('publish')} disabled={saving}>
   234→   234→              <Send className="h-4 w-4 mr-1.5" /> {form.status === 'PUBLISHED' ? 'Update' : 'Publish'}
   235→   235→            </Button>
   236→   236→          )}
   237→   237→        </div>
   238→   238→      </div>
   239→   239→
   240→   240→      <div className="grid gap-6 lg:grid-cols-12">
   241→   241→        {/* MAIN COLUMN */}
   242→   242→        <div className="lg:col-span-8 space-y-5">
   243→   243→          {/* Title + slug — WP style */}
   244→   244→          <div className="rounded-xl border border-border bg-card p-5">
   245→   245→            <Label className="text-xs text-muted-foreground">Title</Label>
   246→   246→            <input
   247→   247→              value={form.title}
   248→   248→              onChange={(e) => onTitleChange(e.target.value)}
   249→   249→              placeholder="Add title"
   250→   250→              className="w-full text-3xl font-display font-semibold bg-transparent border-0 outline-none focus:ring-0 px-0 py-1 placeholder:text-muted-foreground/40"
   251→   251→            />
   252→   252→            {form.slug && (
   253→   253→              <div className="mt-1 flex items-center gap-1 text-xs text-muted-foreground">
   254→   254→                <span className="opacity-60">Permalink:</span>
   255→   255→                <span className="font-mono">/{form.slug}</span>
   256→   256→                <button onClick={() => { slugTouched.current = true }} className="ml-1 text-primary hover:underline">Edit</button>
   257→   257→              </div>
   258→   258→            )}
   259→   259→            {slugTouched.current && (
   260→   260→              <Input
   261→   261→                value={form.slug}
   262→   262→                onChange={(e) => update({ slug: slugify(e.target.value) })}
   263→   263→                onBlur={() => { slugTouched.current = false }}
   264→   264→                placeholder="url-slug"
   265→   265→                className="font-mono text-sm mt-2"
   266→   266→                autoFocus
   267→   267→              />
   268→   268→            )}
   269→   269→          </div>
   270→   270→
   271→   271→          {/* Content editor — WP classic style */}
   272→   272→          <div className="rounded-xl border border-border bg-card overflow-hidden">
   273→   273→            <div className="flex items-center justify-between px-4 py-2.5 border-b border-border bg-secondary/40">
   274→   274→              <div className="flex items-center gap-2">
   275→   275→                <FileText className="h-4 w-4 text-muted-foreground" />
   276→   276→                <span className="text-sm font-medium">Content</span>
   277→   277→                <span className="text-xs text-muted-foreground">· Markdown</span>
   278→   278→              </div>
   279→   279→              <div className="flex items-center gap-2 text-xs text-muted-foreground flex-wrap">
   280→   280→                <span className="hidden sm:inline">{wordCount} words</span>
   281→   281→                <span className="hidden sm:inline">·</span>
   282→   282→                <span className="hidden sm:inline">{estimateReadTime(form.content)} min read</span>
   283→   283→                <Button variant="ghost" size="sm" className="h-7" onClick={() => setShowGenerator(true)}>
   284→   284→                  <Wand2 className="h-3.5 w-3.5 mr-1 text-primary" /> AI Image
   285→   285→                </Button>
   286→   286→                <Button variant="ghost" size="sm" className="h-7" onClick={() => setShowAffiliateDialog(true)} disabled={form.affiliateLinks.length === 0}>
   287→   287→                  <ShoppingCart className="h-3.5 w-3.5 mr-1 text-primary" /> Affiliate
   288→   288→                </Button>
   289→   289→                <Button variant="ghost" size="sm" className="h-7" onClick={() => setShowHtmlDialog(true)}>
   290→   290→                  <Code2 className="h-3.5 w-3.5 mr-1 text-primary" /> HTML
   291→   291→                </Button>
   292→   292→              </div>
   293→   293→            </div>
   294→   294→            <Tabs defaultValue="write">
   295→   295→              <div className="px-4 pt-2 border-b border-border">
   296→   296→                <TabsList className="bg-transparent h-9">
   297→   297→                  <TabsTrigger value="write" className="text-sm"><Sparkles className="h-3.5 w-3.5 mr-1.5" /> Visual</TabsTrigger>
   298→   298→                  <TabsTrigger value="preview" className="text-sm"><Eye className="h-3.5 w-3.5 mr-1.5" /> Preview</TabsTrigger>
   299→   299→                </TabsList>
   300→   300→              </div>
   301→   301→              <TabsContent value="write" className="m-0">
   302→   302→                <Textarea
   303→   303→                  ref={contentRef}
   304→   304→                  value={form.content}
   305→   305→                  onChange={(e) => update({ content: e.target.value })}
   306→   306→                  placeholder={'Start writing…\n\n## Use Markdown for formatting\n\n- **bold**, *italic*, > quotes\n- ### subheadings\n- lists\n\nSeparate paragraphs with blank lines.'}
   307→   307→                  className="font-mono text-sm min-h-[480px] resize-y border-0 rounded-none focus-visible:ring-0"
   308→   308→                />
   309→   309→              </TabsContent>
   310→   310→              <TabsContent value="preview" className="m-0">
   311→   311→                <div className="min-h-[480px] p-6">
   312→   312→                  {form.content ? <Markdown content={form.content} /> : <p className="text-muted-foreground text-sm">Nothing to preview yet.</p>}
   313→   313→                </div>
   314→   314→              </TabsContent>
   315→   315→            </Tabs>
   316→   316→          </div>
   317→   317→
   318→   318→          {/* Excerpt metabox (full width, below content like WP) */}
   319→   319→          <Metabox title="Excerpt" icon={<FileText className="h-4 w-4" />} description="A short summary shown in cards, search results and SEO previews. Leave blank to auto-generate from content.">
   320→   320→            <Textarea
   321→   321→              value={form.excerpt}
   322→   322→              onChange={(e) => update({ excerpt: e.target.value })}
   323→   323→              placeholder="Optional hand-written summary…"
   324→   324→              rows={3}
   325→   325→              className="resize-none"
   326→   326→            />
   327→   327→            <p className="text-[10px] text-muted-foreground mt-1">{form.excerpt.length} characters</p>
   328→   328→          </Metabox>
   329→   329→
   330→   330→          {/* Affiliate links metabox (full width) */}
   331→   331→          <Metabox title="Affiliate Links" icon={<ShoppingCart className="h-4 w-4" />} description="Recommended products, supplies or books. Displayed at the bottom of the article with a clear disclosure." action={<Badge variant="secondary">{form.affiliateLinks.length}</Badge>}>
   332→   332→            <AffiliateLinksEditor links={form.affiliateLinks} onChange={(links) => update({ affiliateLinks: links })} />
   333→   333→          </Metabox>
   334→   334→        </div>
   335→   335→
   336→   336→        {/* SIDEBAR COLUMN — WordPress metaboxes */}
   337→   337→        <aside className="lg:col-span-4 space-y-4">
   338→   338→          {/* Publish metabox — the heart of WP */}
   339→   339→          <Metabox title="Publish" icon={<Send className="h-4 w-4" />} defaultOpen>
   340→   340→            <div className="space-y-4">
   341→   341→              {/* Status / visibility */}
   342→   342→              <div>
   343→   343→                <Label className="text-xs text-muted-foreground mb-1.5 block">Publish action</Label>
   344→   344→                <RadioGroup
   345→   345→                  value={form.status === 'PUBLISHED' ? 'PUBLISHED' : form.status === 'SCHEDULED' ? 'SCHEDULED' : 'DRAFT'}
   346→   346→                  onValueChange={(v) => update({ status: v })}
   347→   347→                  className="space-y-1.5"
   348→   348→                >
   349→   349→                  <label className="flex items-center gap-2 text-sm cursor-pointer rounded-md px-2 py-1.5 hover:bg-accent">
   350→   350→                    <RadioGroupItem value="DRAFT" /> <Save className="h-3.5 w-3.5 text-muted-foreground" /> Save as Draft
   351→   351→                  </label>
   352→   352→                  <label className="flex items-center gap-2 text-sm cursor-pointer rounded-md px-2 py-1.5 hover:bg-accent">
   353→   353→                    <RadioGroupItem value="PUBLISHED" /> <Send className="h-3.5 w-3.5 text-muted-foreground" /> Publish now
   354→   354→                  </label>
   355→   355→                  <label className="flex items-center gap-2 text-sm cursor-pointer rounded-md px-2 py-1.5 hover:bg-accent">
   356→   356→                    <RadioGroupItem value="SCHEDULED" /> <Clock className="h-3.5 w-3.5 text-muted-foreground" /> Schedule for later
   357→   357→                  </label>
   358→   358→                </RadioGroup>
   359→   359→              </div>
   360→   360→
   361→   361→              {/* Schedule date — only relevant for SCHEDULED, but always editable */}
   362→   362→              <div>
   363→   363→                <Label className="text-xs text-muted-foreground mb-1.5 flex items-center gap-1.5">
   364→   364→                  <Calendar className="h-3.5 w-3.5" /> {form.status === 'SCHEDULED' ? 'Scheduled for' : form.status === 'PUBLISHED' ? 'Publish date' : 'Publish date (optional)'}
   365→   365→                </Label>
   366→   366→                <Input
   367→   367→                  type="datetime-local"
   368→   368→                  value={form.publishedAt}
   369→   369→                  onChange={(e) => update({ publishedAt: e.target.value })}
   370→   370→                />
   371→   371→                {form.status === 'SCHEDULED' && scheduledDate && (
   372→   372→                  <p className={`text-[11px] mt-1 flex items-center gap-1 ${isScheduledFuture ? 'text-amber-600' : 'text-destructive'}`}>
   373→   373→                    <AlertCircle className="h-3 w-3" />
   374→   374→                    {isScheduledFuture
   375→   375→                      ? `Will publish ${scheduledDate.toLocaleString([], { dateStyle: 'long', timeStyle: 'short' })}`
   376→   376→                      : 'Date is in the past — pick a future date to schedule.'}
   377→   377→                  </p>
   378→   378→                )}
   379→   379→                {form.status === 'SCHEDULED' && !form.publishedAt && (
   380→   380→                  <p className="text-[11px] mt-1 text-amber-600 flex items-center gap-1">
